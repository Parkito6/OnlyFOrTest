#include <stdio.h>
int main(void){
    int req;
    int guess;
    int attemp = 1;
    scanf("%d", &req);
    scanf("%d", &guess);
    while (guess != req){
        if(guess<req){
            printf("it is more\n");
        }
        if(guess>req){
            printf("it is less\n");
        }
        attemp = attemp + 1;
        scanf("%d", &guess);
    }
    printf("Number of tries needed:\n");
    printf("%d", attemp);
    return 0;
}
