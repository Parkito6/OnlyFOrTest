#include <stdio.h>
int main(void){
    int i;
    int l;
    int nbwords;
    int max = 0;
    char word[101];
    scanf("%d", &nbwords);
    for (i=0; i<nbwords; i++){
        scanf("%s", word);
        l = 0;
        while (word[l] != '\0'){
            l++;
        }
        if (l>max){
            max = l;
        }
    }
    printf("%d", max);
    return 0;
}
